import numpy as np

def select_next_node(current_node: int, destination_node: int, unvisited_nodes: set, distance_matrix: np.ndarray) -> int:
    """ 
    Select the next node to visit from the unvisited nodes using the greedy algorithm.
    The greedy algorithm selects the unvisited node that has the shortest distance to the current node.

    Parameters
    ----------
    current_node : int
        The current node.
    destination_node : int
        The destination node to return to.
    unvisited_nodes : set
        The set of unvisited nodes.
    distance_matrix : np.ndarray
        The distance matrix of shape (n_nodes, n_nodes).

    Returns
    -------
    nxt_node : int
        The next node to visit.
    """
    min_dist = np.inf
    nxt_node = None
    
    for node in unvisited_nodes:
        dist = distance_matrix[current_node][node]
        
        if dist < min_dist:
            min_dist = dist
            nxt_node = node
    
    return nxt_node
